> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add structured message rendering (syntax-highlighted code, tables, tool-call bubbles, reasoning accordion), a persistent Canvas panel for AI-generated artifacts, and long-press message interactions — all on the HERALD-rebranded codebase.

**Architecture:** Parse-on-render (Option 1). Messages stay as raw strings. `MarkdownParser.swift`'s existing `MarkdownSegment` enum is extended with `.thinking`, `.toolCall`, and `.table` cases. Block renderers live under `Herald/Features/Chat/Renderers/`. Canvas is a new `Herald/Features/Canvas/` module. Long-press attaches `.contextMenu` to `MessageBubble`. No relay or connector changes.

**Tech Stack:** Swift 6.2 / SwiftUI, iOS 26+. No new third-party dependencies.

**Prerequisites:** HERALD rebrand plan (`2026-07-19-herald-rebrand.md`) must be complete first. All file paths below use `Herald/` (post-rebrand naming).

- No new Swift Package dependencies — syntax highlighting is a ~150-keyword in-file tokenizer

```swift
// Replace <think>…</think> spans with a sentinel so the fenced-block scanner skips them,

// Simpler approach: scan for <think> tags first, split on them:
private func extractThinkingBlocks(_ text: String) -> [MarkdownSegment] {
    var segments: [MarkdownSegment] = []
    var remainder = text
    let openTag = "<think>"
    let closeTag = "</think>"
    
    while let openRange = remainder.range(of: openTag) {
        // Prose before the tag
        let before = String(remainder[..<openRange.lowerBound])
        if !before.isEmpty {
            // Further parse this prose for code blocks / images
            segments += parseMarkdownWithoutThinking(before)
        }
        remainder = String(remainder[openRange.upperBound...])
        // Find closing tag
        if let closeRange = remainder.range(of: closeTag) {
            let thinkContent = String(remainder[..<closeRange.lowerBound])
            segments.append(.thinking(content: thinkContent))
            remainder = String(remainder[closeRange.upperBound...])
        } else {
            // Unclosed think tag — treat rest as thinking
            segments.append(.thinking(content: remainder))
            remainder = ""
        }
    }
    // Remaining text after last </think>
    if !remainder.isEmpty {
        segments += parseMarkdownWithoutThinking(remainder)
    }
    return segments
}
```

- [ ] **Step 3: Add table parsing inside parseMarkdownWithoutThinking**

In the prose-handling path (after splitting on fenced blocks and images), detect markdown table lines. A table section is a consecutive run of lines starting with `|`:

```swift
private func extractTables(from prose: String) -> [MarkdownSegment] {
    var result: [MarkdownSegment] = []
    var lines = prose.components(separatedBy: "\n")
    var i = 0
    var textAccumulator: [String] = []
    
    while i < lines.count {
        let line = lines[i]
        if line.trimmingCharacters(in: .whitespaces).hasPrefix("|") {
            // Flush accumulated prose
            let text = textAccumulator.joined(separator: "\n")
            if !text.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty {
                result += splitProseAndImages(text)
            }
            textAccumulator = []
            // Collect table lines
            var tableLines: [String] = []
            while i < lines.count && lines[i].trimmingCharacters(in: .whitespaces).hasPrefix("|") {
                tableLines.append(lines[i])
                i += 1
            }
            // Parse rows (skip separator lines: |---|---|)
            let rows = tableLines.compactMap { line -> [String]? in
                let cells = line.components(separatedBy: "|")
                    .dropFirst().dropLast()
                    .map { $0.trimmingCharacters(in: .whitespaces) }
                // Skip separator rows (all dashes)
                if cells.allSatisfy({ $0.allSatisfy({ $0 == "-" || $0 == ":" || $0 == " " }) }) {
                    return nil
                }
                return Array(cells)
            }
            if !rows.isEmpty {
                result.append(.table(rows: rows))
            }
        } else {
            textAccumulator.append(line)
            i += 1
        }
    }
    let remaining = textAccumulator.joined(separator: "\n")
    if !remaining.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty {
        result += splitProseAndImages(remaining)
    }
    return result
}
```

Replace the `splitProseAndImages` call inside `parseMarkdownWithoutThinking`'s prose handling with `extractTables`.

- [ ] **Step 4: Add the toolActivities overload**

```swift
// New public entry point used by MarkdownContentView when a message has tool events
func parseMarkdown(_ text: String, toolActivities: [ToolActivity]) -> [MarkdownSegment] {
    var segments = parseMarkdown(text)
    
    // Append tool-call segments after the message content
    // (tool calls are emitted by the relay as separate SSE events, not inline in the text)
    let toolSegments = toolActivities.map { activity -> MarkdownSegment in
        // ToolActivity.label has format "tool_name(args)" or just "tool_name"
        // Split on first "(" to separate name from args
        let label = activity.label
        if let parenIdx = label.firstIndex(of: "(") {
            let name = String(label[..<parenIdx])
            let argsWithParen = String(label[parenIdx...])
            return .toolCall(name: name, args: argsWithParen, result: nil)
        } else {
            return .toolCall(name: label, args: nil, result: nil)
        }
    }
    // Insert tool segments before the final prose if any tool activities exist
    if !toolSegments.isEmpty {
        // Find position to insert: before the last prose segment, or at the front
        if let lastProseIdx = segments.lastIndex(where: {
            if case .prose = $0 { return true }
            return false
        }) {
            segments.insert(contentsOf: toolSegments, at: lastProseIdx)
        } else {
            segments = toolSegments + segments
        }
    }
    return segments
}
```

```swift
struct SyntaxHighlighter {

    static let keywordColor   = Color(hex: 0xFF6B00)   // molten orange

                   "await", "raise", "del", "global", "nonlocal", "assert"],
        "javascript": ["function", "const", "let", "var", "return", "if", "else",
                       "for", "while", "class", "import", "export", "default",
                       "async", "await", "try", "catch", "throw", "new", "this",

                       "switch", "case", "break", "continue", "from", "of"],

                       "for", "while", "class", "import", "export", "default",
                       "async", "await", "try", "catch", "throw", "new", "this",
                       "null", "undefined", "true", "false", "type", "interface",
                       "enum", "namespace", "extends", "implements", "readonly",
                       "private", "public", "protected", "abstract", "switch"],
        "bash": ["if", "then", "else", "elif", "fi", "for", "do", "done",
                 "while", "case", "esac", "function", "return", "echo", "export",
                 "local", "readonly", "shift", "exit", "source", "true", "false"],
        "sql": ["SELECT", "FROM", "WHERE", "JOIN", "LEFT", "RIGHT", "INNER",
                "OUTER", "ON", "GROUP", "BY", "ORDER", "HAVING", "INSERT",
                "INTO", "VALUES", "UPDATE", "SET", "DELETE", "CREATE", "TABLE",
                "DROP", "ALTER", "ADD", "COLUMN", "INDEX", "PRIMARY", "KEY",
                "FOREIGN", "REFERENCES", "NOT", "NULL", "UNIQUE", "DEFAULT",
                "AND", "OR", "IN", "LIKE", "BETWEEN", "AS", "DISTINCT",
                "COUNT", "SUM", "AVG", "MAX", "MIN", "WITH", "UNION"],
    ]

    /// Returns an AttributedString with syntax coloring for the given language.
    /// Falls back to plain off-white text for unknown languages.
    static func tokenize(code: String, language: String?) -> AttributedString {
        let lang = language?.lowercased() ?? ""
        guard let kwSet = keywords[lang] else {
            var attr = AttributedString(code)
            attr.foregroundColor = defaultColor
            return attr
        }

        var result = AttributedString()
        // Tokenize by splitting on word boundaries and whitespace/punctuation
        let tokens = tokenize(raw: code)
        for token in tokens {
            var attr = AttributedString(token)
            if kwSet.contains(token) {
                attr.foregroundColor = keywordColor
            } else if token.hasPrefix("//") || token.hasPrefix("#") {
                attr.foregroundColor = commentColor
            } else if token.hasPrefix("\"") || token.hasPrefix("'") || token.hasPrefix("`") {
                attr.foregroundColor = stringColor
            } else if Double(token) != nil {
                attr.foregroundColor = numberColor
            } else {
                attr.foregroundColor = defaultColor
            }
            result += attr
        }
        return result
    }

    private static func tokenize(raw: String) -> [String] {
        // Split preserving delimiters — simple character-by-character scan
        var tokens: [String] = []
        var current = ""
        var i = raw.startIndex
        while i < raw.endIndex {
            let ch = raw[i]
            if ch.isLetter || ch == "_" || ch.isNumber && !current.isEmpty {
                current.append(ch)
            } else {
                if !current.isEmpty {
                    tokens.append(current)
                    current = ""
                }
                tokens.append(String(ch))
            }
            i = raw.index(after: i)
        }
        if !current.isEmpty { tokens.append(current) }
        return tokens
    }
}
```

```swift
struct MarkdownContentView: View {
    let content: String
    let isStreaming: Bool
    var toolActivities: [ToolActivity] = []

    @State private var cachedSegments: [MarkdownSegment] = []
    @State private var lastContent: String = ""

    private var segments: [MarkdownSegment] {
        // Recompute only when content changes
        if content != lastContent {
            return parseMarkdown(content, toolActivities: toolActivities)
        }
        return cachedSegments
    }

    var body: some View {
        VStack(alignment: .leading, spacing: Design.Spacing.xs) {
            ForEach(computedSegments) { segment in
                segmentView(segment)
            }
        }
        .onChange(of: content) { _, newValue in
            cachedSegments = parseMarkdown(newValue, toolActivities: toolActivities)
            lastContent = newValue
        }
        .onAppear {
            if cachedSegments.isEmpty {
                cachedSegments = parseMarkdown(content, toolActivities: toolActivities)
                lastContent = content
            }
        }
    }

    private var computedSegments: [MarkdownSegment] {
        cachedSegments.isEmpty ? parseMarkdown(content, toolActivities: toolActivities) : cachedSegments
    }
```

```swift
import Foundation

enum HeraldArtifactType: String, Codable {
    case code(language: String)
    case markdown
    case svg

    // Custom Codable since associated values aren't Codable by default
    private enum CodingKeys: String, CodingKey { case type, language }

    init(from decoder: Decoder) throws {
        let c = try decoder.container(keyedBy: CodingKeys.self)
        let type = try c.decode(String.self, forKey: .type)
        switch type {
        case "code":
            let lang = try c.decodeIfPresent(String.self, forKey: .language) ?? ""
            self = .code(language: lang)
        case "svg":
            self = .svg
        default:
            self = .markdown
        }
    }

    func encode(to encoder: Encoder) throws {
        var c = encoder.container(keyedBy: CodingKeys.self)
        switch self {
        case .code(let lang):
            try c.encode("code", forKey: .type)
            try c.encode(lang, forKey: .language)
        case .markdown:
            try c.encode("markdown", forKey: .type)
        case .svg:
            try c.encode("svg", forKey: .type)
        }
    }

    var displayName: String {
        switch self {
        case .code(let lang): return lang.isEmpty ? "Code" : lang
        case .markdown: return "Markdown"
        case .svg: return "SVG"
        }
    }
}

struct HeraldArtifact: Codable, Identifiable {
    var id: UUID
    var sessionID: String
    var type: HeraldArtifactType
    var content: String
    var updatedAt: Date

    init(sessionID: String, type: HeraldArtifactType, content: String) {
        self.id = UUID()
        self.sessionID = sessionID
        self.type = type
        self.content = content
        self.updatedAt = Date()
    }
}
```

- [ ] **Step 2: Create HeraldCanvasStore.swift**

```swift
import SwiftUI
import Foundation

@Observable
@MainActor
final class HeraldCanvasStore {
    var activeArtifact: HeraldArtifact?

    private let defaults = UserDefaults.standard
    private let storageKey = "herald.canvas.artifacts"

    // Load the stored artifact for a given session on startup
    func loadArtifact(for sessionID: String) {
        guard let data = defaults.data(forKey: storageKey + "." + sessionID),
              let artifact = try? JSONDecoder().decode(HeraldArtifact.self, from: data)
        else { return }
        activeArtifact = artifact
    }

    // Open an artifact extracted from a message
    func open(message: Message) {
        let segments = parseMarkdown(message.content)
        guard let codeBlock = segments.first(where: {
            if case .codeBlock = $0 { return true }
            return false
        }), case .codeBlock(_, let lang, let code) = codeBlock else { return }

        let type: HeraldArtifactType = (lang?.lowercased() == "svg") ? .svg
            : lang.map { .code(language: $0) } ?? .markdown
        let artifact = HeraldArtifact(
            sessionID: message.conversationID ?? "unknown",
            type: type,
            content: code
        )
        activeArtifact = artifact
        persist(artifact)
    }

    // Called from CanvasView when user edits content
    func updateContent(_ newContent: String) {
        guard var artifact = activeArtifact else { return }
        artifact.content = newContent
        artifact.updatedAt = Date()
        activeArtifact = artifact
        persist(artifact)
    }

    func clear() {
        if let sessionID = activeArtifact?.sessionID {
            defaults.removeObject(forKey: storageKey + "." + sessionID)
        }
        activeArtifact = nil
    }

    private func persist(_ artifact: HeraldArtifact) {
        if let data = try? JSONEncoder().encode(artifact) {
            defaults.set(data, forKey: storageKey + "." + artifact.sessionID)
        }
    }
}
```
